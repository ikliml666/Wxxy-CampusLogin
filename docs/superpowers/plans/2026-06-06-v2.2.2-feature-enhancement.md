### Task A2: 创建i18n初始化配置

**Files:**
- Create: `frontend/src/i18n/index.ts`

- [ ] **Step 1: 创建i18n初始化文件**

```typescript
// frontend/src/i18n/index.ts
import i18n from 'i18next'
import { initReactI18next } from 'react-i18next'
import LanguageDetector from 'i18next-browser-languagedetector'

import zh from './locales/zh.json'
import en from './locales/en.json'

i18n
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    resources: {
      zh: { translation: zh },
      en: { translation: en },
    },
    fallbackLng: 'zh',
    lng: localStorage.getItem('app-language') || undefined,
    detection: {
      order: ['localStorage', 'navigator'],
      lookupLocalStorage: 'app-language',
      caches: ['localStorage'],
    },
    interpolation: {
      escapeValue: false,
    },
  })

export default i18n
```

```bash
git add frontend/src/i18n/index.ts

fn cleanup_old_logs(log_dir: &Path, retention_days: u32) {
    if retention_days == 0 {
        return; // 永久保存
    }
    let cutoff = std::time::SystemTime::now()
        - std::time::Duration::from_secs(retention_days as u64 * 86400);
    if let Ok(entries) = std::fs::read_dir(log_dir) {
        for entry in entries.flatten() {
            if let Ok(metadata) = entry.metadata() {
                if let Ok(modified) = metadata.modified() {
                    if modified < cutoff {
                        let _ = std::fs::remove_file(entry.path());
                    }
                }
            }
        }
    }
}
```

- [ ] **Step 2: 在logger_worker中添加定时清理**

在 `logger_worker` 函数的 channel 接收循环中，添加定时清理逻辑。使用 `recv_timeout` 替代 `recv`，每3600秒（1小时）触发一次清理：

const CLEANUP_INTERVAL: std::time::Duration = std::time::Duration::from_secs(3600);

loop {
    match rx.recv_timeout(std::time::Duration::from_secs(60)) {
        Ok(msg) => {
            // ... 处理现有消息逻辑 ...
        }
        Err(std::sync::mpsc::RecvTimeoutError::Timeout) => {

                if let Some(dir) = log_dir.as_ref() {
                    let retention_days = get_log_retention_days();
                    cleanup_old_logs(dir, retention_days);
                }
                last_cleanup = std::time::Instant::now();
            }
        }
        Err(std::sync::mpsc::RecvTimeoutError::Disconnected) => break,
    }
}

import { useEffect, useRef } from 'react'
import { gsap } from 'gsap'

interface PulseOptions {
  type: 'heartbeat' | 'statusPulse' | 'loadingPulse'
  duration?: number
  maxScale?: number
}

export function usePulseAnimation(options: PulseOptions) {
  const ref = useRef<HTMLElement>(null)
  const { type, duration, maxScale } = options

  useEffect(() => {
    const el = ref.current
    if (!el) return

    let tween: gsap.core.Tween

    switch (type) {
      case 'heartbeat':
        tween = gsap.to(el, {
          scale: 1.25,
          duration: duration ?? 0.3,
          ease: 'power2.out',
          yoyo: true,
          repeat: 1,
          onComplete: () => {
            gsap.to(el, {
              scale: 1.15,
              duration: 0.15,
              yoyo: true,
              repeat: 1,
              ease: 'power2.inOut',
            })
          },
          force3D: true,
        })
        // 循环
        const timeline = gsap.timeline({ repeat: -1, repeatDelay: 2.4 })
        timeline.to(el, { scale: 1.25, duration: 0.3, ease: 'power2.out' })
        timeline.to(el, { scale: 1, duration: 0.15, ease: 'power2.in' })
        timeline.to(el, { scale: 1.15, duration: 0.15, ease: 'power2.out' })
        timeline.to(el, { scale: 1, duration: 0.15, ease: 'power2.in' })
        return () => { timeline.kill() }

      case 'statusPulse':
        tween = gsap.to(el, {
          scale: maxScale ?? 1.1,
          opacity: 0.7,
          duration: duration ?? 0.3,
          ease: 'power2.out',
          yoyo: true,
          repeat: 1,
          force3D: true,
        })
        const tl = gsap.timeline({ repeat: 1 })
        tl.to(el, { scale: 1.1, opacity: 0.7, duration: 0.3, force3D: true })
        tl.to(el, { scale: 1, opacity: 1, duration: 0.3, force3D: true })
        tl.to(el, { scale: 1.1, opacity: 0.7, duration: 0.3, force3D: true })
        tl.to(el, { scale: 1, opacity: 1, duration: 0.3, force3D: true })
        return () => { tl.kill() }

      case 'loadingPulse':
        const loadingTl = gsap.timeline({ repeat: -1 })
        loadingTl.to(el, { opacity: 0.5, scale: 0.95, duration: 0.6, ease: 'sine.inOut', force3D: true })
        loadingTl.to(el, { opacity: 1, scale: 1, duration: 0.6, ease: 'sine.inOut', force3D: true })
        return () => { loadingTl.kill() }
    }
  }, [type, duration, maxScale])

  return ref
}
